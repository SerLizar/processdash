[Size Estimating Template Tag] = TAG;

/*
 * Dynamic lists of user-entered data.
 */

[Est_Base_Additions]     = search("Base Additions", "LOC", 1);
[Est_New_Objects]        = search("New Objects",    "LOC", 1);
[Est_Reused_Objects]     = search("Reused Objects", "LOC", 1);
[Est_New_Reused_Objects] = search("New Objects",    "LOC", [New Reused?]);

[Base_Additions]     = search("Base Additions", "Actual LOC", 1);
[New_Objects]        = search("New Objects",    "Actual LOC", 1);
[Reused_Objects]     = search("Reused Objects", "Actual LOC", 1);
[New_Reused_Objects] = search("New Objects",    "Actual LOC",
                              [Actual New Reused?]);


/*
 * Data Elements to track Estimated LOC.
 */

[Estimated New & Changed LOC] = ?NaN;

[Estimated Base LOC]     = 0;
[Estimated Deleted LOC]  = 0;
[Estimated Modified LOC] = 0;

[Estimated Base Additions LOC] = sumFor("LOC", [Est_Base_Additions]);
[Estimated New Objects LOC]    = sumFor("LOC", [Est_New_Objects]);
[Estimated Reused LOC]	       = sumFor("LOC", [Est_Reused_Objects]);
[Estimated New Reused LOC]     = sumFor("LOC", [Est_New_Reused_Objects]);

[Estimated Object LOC] = [Estimated Base Additions LOC] +
                         [Estimated New Objects LOC] +
			 [Estimated Modified LOC];
[Estimated Added LOC] =
        [Estimated New & Changed LOC] - [Estimated Modified LOC];

[Estimated Total LOC] = [Estimated Base LOC] + [Estimated Reused LOC]
        + [Estimated New & Changed LOC] - [Estimated Modified LOC]
        - [Estimated Deleted LOC];

[Estimated Max LOC] = 0;
[Estimated Min LOC] = 0;
[Estimated New & Changed KLOC] = [Estimated New & Changed LOC] / 1000;
[Estimated New & Changed LOC/Beta0] = 0;
[Estimated New & Changed LOC/Beta1] = 0;
[Estimated New & Changed LOC/Range] = 0;
[Estimated Total LOC/Range] = [Estimated New & Changed LOC/Range];
[Estimated New & Changed LOC/Interval Percent] = 0;
[Estimated New & Changed LOC/Probe Method] = "D";
[Estimated New & Changed LOC/R Squared] = 0;

/*
 * Data Elements to track Actual LOC
 */

[Base LOC]     = 0;
[Deleted LOC]  = 0;
[Modified LOC] = 0;
[Total LOC]    = ?NaN;

[New & Changed LOC] = [Added LOC] + [Modified LOC];
[Added LOC] = [Total LOC] - [Base LOC] + [Deleted LOC] - [Reused LOC];

[Base Additions LOC] = sumFor("Actual LOC", [Base_Additions]);
[New Objects LOC]    = sumFor("Actual LOC", [New_Objects]);
[Reused LOC]         = sumFor("Actual LOC", [Reused_Objects]);
[New Reused LOC]     = sumFor("Actual LOC", [New_Reused_Objects]);

[Object LOC] = [Base Additions LOC] + [Modified LOC] + [New Objects LOC];

[New & Changed KLOC] = [New & Changed LOC] / 1000;

/*
 * Data elements for PROBE-estimated time values
 */

[Estimated Max Time] = 0;
[Estimated Min Time] = 0;

[Estimated Time/Beta0] = 0;
[Estimated Time/Beta1] = 0;
[Estimated Time/Range] = 0;
[Estimated Time/Interval Percent] = 0;
[Estimated Time/Probe Method] = "D";
[Estimated Time/R Squared] = 0;
